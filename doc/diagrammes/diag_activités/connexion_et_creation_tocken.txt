**DIAGRAMME D'ACTIVITE DE CONNEXION ET CREATION DE TOCKEN

@startuml
start

:Utilisateur (id, mot de passe);
:Vérifier si l'utilisateur existe en base;

if (Utilisateur existe ?) then (oui)
    :Vérifier si un token valide existe déjà;
    if (Token valide trouvé ?) then (oui)
        :Retourner le token existant;
    else (non)
        :Générer un nouveau token;
        :Enregistrer le token en base;
        :Retourner le nouveau token;
    endif
else (non)
    : message d'erreur\n"Veuillez créer un compte";
endif

stop
@enduml
