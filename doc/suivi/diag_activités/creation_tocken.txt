**DIAGRAMME D'ACTIVITE DE CREATION DE TOCKEN

@startuml
start

:Utilisateur (id, mot de passe);
:Vérifier si l'utilisateur existe en base;

if (Utilisateur existe ?) then (oui)
    :Vérifier si un token valide existe déjà;
    if (Token valide trouvé ?) then (oui)
        :Retourner le token existant;
    else (non)
        :Générer un nouveau token;
        :Enregistrer le token en base;
        :Retourner le nouveau token;
    endif
else (non)
    : message d'erreur\n"Veuillez créer un compte";
endif

stop
@enduml

** DIAGRAMME D'ACTIVITE DE CREATION D'UN Qrcode

@startuml

start
:Demande de création de QR code(tocken,
contenu, expiry_date, type,usage_limit);


:Vérifier le token de l'utilisateur?;
if (Token valide ?) then (oui)
  :Générer le QR code;
  :Enregistrer le QR code 
  :Retourner le QR code à l'utilisateur;
else (non)
  :Renvoyer message d'erreur ou invitation à créer un compte;
endif


stop
@enduml
